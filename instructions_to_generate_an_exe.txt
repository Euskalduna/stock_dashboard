# Instructions
1- install "pyinstaller"
2- use the command
    pyi-makespec --windowed --name <name_of_the_distribution> <name_of_the_run.py>  # el --windowed es para que no salga la consola
OR
    pyi-makespec --console --name <name_of_the_distribution> <name_of_the_run.py>   # el --console es para que salga la consola
EXAMPLE:
    pyi-makespec --windowed --name Cartera_Cuadro_de_Mando my_main.py
OR
    pyi-makespec --console --name Cartera_Cuadro_de_Mando my_main.py


3- The previous command will generate an .spec file (<name_of_the_distribution>.spec) in the same directory
Go to that file and edit it as follows:

    Add the following import at the begining of the file as it is needed for including dash_pivottable
        import os
        import dash_pivottable
        pivottable_path = os.path.dirname(dash_pivottable.__file__)

    Edit the following parameters in .spec file:
        pathex=['C:\\Users\\<user_name>\\PycharmProjects\\stock_dashboard'],    --> put the base path of the project
        datas=[
            ('assets','assets'),
            ('configs','configs'),
            ('pages','pages'),
            ('utils','utils'),
            (pivottable_path, 'dash_pivottable'),
        ],                                                              --> put the folders that need to be included
        hiddenimports=[
            'pkg_resources.py2_warn',
            'pandas',
            'plotly.graph_objects',
            'dash_core_components',
            'dash_html_components',
            'dash_table',
            'forex_python',
            'forex_python.converter',
            'yfinance',
            'dash_pivottable'
        ],                                                              --> these are the imports to take into account
                                                                        basically, you will know what to include
                                                                        whenever you generate the .exe and it fails because
                                                                        it doesnt have the X library
4- Generate the .exe using the command
    pyinstaller <name_of_the_distribution>.spec --noconfirm
EXAMPLE:
    pyinstaller Cartera_Cuadro_de_Mando.spec --noconfirm

5- Copy the "data" folder of the project in the following path:
    /dist/<name_of_the_distribution>
EXAMPLE:
    /dist/Cartera_Cuadro_de_Mando

# -*- mode: python ; coding: utf-8 -*-


a = Analysis(
    ['my_main.py'],
    pathex=['C:\\Users\\Ale\\PycharmProjects\\stock_dashboard'],
    binaries=[],
    datas=[
        ('assets','assets'),
        ('pages','pages'),
        ('utils','utils'),
    ],
    hiddenimports=[
        'pkg_resources.py2_warn',
        'pandas',
        'plotly.graph_objects',
        'dash_core_components',
        'dash_html_components',
        'dash_table',
        'forex_python',
        'forex_python.converter',
        'yfinance',
    ],
    hookspath=[],
    hooksconfig={},
    runtime_hooks=[],
    excludes=[],
    noarchive=False,
    optimize=0,
)
pyz = PYZ(a.pure)

exe = EXE(
    pyz,
    a.scripts,
    [],
    exclude_binaries=True,
    name='Cartera_Cuadro_de_Mando',
    debug=True,
    bootloader_ignore_signals=False,
    strip=False,
    upx=True,
    console=False,
    disable_windowed_traceback=False,
    argv_emulation=False,
    target_arch=None,
    codesign_identity=None,
    entitlements_file=None,
)
coll = COLLECT(
    exe,
    a.binaries,
    a.datas,
    strip=False,
    upx=True,
    upx_exclude=[],
    name='Cartera_Cuadro_de_Mando',
)
